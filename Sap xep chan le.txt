#include <stdio.h>
int InterchangeSort(int a[], int n){
	for(int i=0;i<n-1;i++){
		for(int j=i+1;j<n;j++){
			if(a[i]>a[j]){
				int tmp=a[i];
				a[i]=a[j];
				a[j]=tmp;
			}
		}
	}
}
void xuat(int a[], int n){
	for(int i=0; i<n; i++){
		printf("%d ",a[i]);
	}
}
