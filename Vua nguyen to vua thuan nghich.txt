#include<stdio.h>
#include<math.h>
int snt(long a){
	if(a<2) return 0;
	else{
	
	int kt=0;
	for(long i=2;i<=sqrt(a);i++){
		if(a%i==0){
		
		kt=1;
		break;	
	}
}
	if(kt==1 ) return 0;
	else return 1;

}
}
int stn(int n){
		int i,s=0,m=n;
	while(n>0){
		int d=n%10;
		s=s*10+d;
		n=n/10;
	}
if(s==m) return 1;
else return 0;
}
int main(){
		int t,k;

		for(i=a;i<=b;i++){
