#include<string.h>

	int t;
	scanf("%d",&t);
	while(t--){
		scanf("\n");

		gets(a);
		int n=strlen(a);
		for(int i=0;i<n;i++){
		    if(a[i]=='0' && a[i+1]=='8'&& a[i+2]=='4'){
		    	b[i]=1;
		    	b[i+1]=1;
		    	b[i+2]=1;
			}
		}
			for(int i=0;i<n;i++){
		    if(b[i]!=1) printf("%c",a[i]);
		}
		for(int i=0;i<n;i++){

		}
		printf("\n");
		
		
		
		
	}
