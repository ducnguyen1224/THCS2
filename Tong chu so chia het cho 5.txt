#include<stdio.h>
#include<math.h>
int nt(int a){
	int i,kt=0;
	if(a<=1) return 0;
	for(i=2;i<=sqrt(a);i++){
		if(a%i==0){
			kt=1;
			break;
		}
	}
	if(kt==1) return 0;
	else return 1;
}
int tcn(int a){
	int s=0;
	while(a>0){
		s=s+a%10;
		a=a/10;
	}
	return s;
}

int main(){
	int n,d=0;
	scanf("%d",&n);
	for(int i=5;i<n;i++){
		if(nt(i)==1 && tcn(i)%5==0){
			d++;
		printf("%d ",i);}
	}
	printf("\n%d",d);
}
