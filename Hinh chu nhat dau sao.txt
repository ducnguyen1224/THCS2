#include<conio.h>
#include<stdio.h>
int main()
{
    int n,m;
    scanf("%d",&n); 
    scanf("%d",&m); 
    for(int i = 1; i <= m; i++)    {
        for(int k = 1 ; k <= n; k++){
            if(i == 1 || k == 1 || i == m || k == n){
                printf("*");
            }
            else{
                printf(" ");
            }
        }
        printf("\n\n");
    }
}
