#include<stdio.h>
#include<math.h>
void nt(long a){
	for(int i=2;i<=sqrt(a);i++){
		int d=0;
		while(a%i==0){
			d++;
			if(i==a) printf(" %d^%d\n",i,d);
			a=a/i;
			if(a%i!=0 && a!=1) printf(" %d^%d *",i,d);
			
		}
	}
	if(a>1) printf(" %d^1\n",a);
}

int main(){
	int t;
    scanf("%d",&t);
    while(t--){
    	long a;
    	scanf("%ld",&a);
    	printf("%ld =",a);
    	nt(a);
}
}
