
#include<stdio.h>
int main(){
	int n;
	scanf("%d",&n);
	int m=2*n-1;
	int x=n,k=0,i,j;
	int a[2*n-1][2*n-1];
	
	while(x>0){
		for(j=k;j<m;j++){
			a[k][j]=x;
		}
		for(i=k+1;i<m;i++){
			a[i][m-1]=x;
		}
		for(j=m-2;j>=k;j--){
			a[m-1][j]=x;
		}
		for(i=m-2;i>k;i--){
			a[i][k]=x;
		}
		m--;
		k++;
		x--;
	}
	for(i=0;i<2*n-1;i++){
		for(j=0;j<2*n-1;j++){
			printf("%d",a[i][j]);
		}
		printf("\n");
	}
}
